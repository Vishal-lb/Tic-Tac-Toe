#include <stdio.h>
#include <stdlib.h>  // For system function

void initializeBoard();
void printBoard();
void playerMove(int player);
int checkWin();
int checkDraw();
void switchPlayer(int *player);

char board[10];

int main() {
    int player = 1, status = -1;

    initializeBoard();
    printBoard();

    while (status == -1) {
        playerMove(player);
        printBoard();

        if (checkWin()) {
            printf("Player %d is the Winner!\n", player);
            return 0;
        }

        if (checkDraw()) {
            printf("It's a draw!\n");
            return 0;
        }

        switchPlayer(&player);
    }

    return 0;
}

void initializeBoard() {
    for (int i = 0; i < 10; i++) {
        board[i] = '0' + i;
    }
}

void printBoard() {
    system("clear");  // Use "cls" on Windows and "clear" on Unix-like systems
    printf("\n\n");
    printf("=== TIC TAC TOE ===\n\n");
    printf("     |     |     \n");
    printf("  %c  |  %c  |  %c  \n", board[1], board[2], board[3]);
    printf("||_\n");
    printf("     |     |     \n");
    printf("  %c  |  %c  |  %c  \n", board[4], board[5], board[6]);
    printf("||_\n");
    printf("     |     |     \n");
    printf("  %c  |  %c  |  %c  \n", board[7], board[8], board[9]);
    printf("     |     |     \n");
    printf("\n\n");
}

void playerMove(int player) {
    int input;
    char mark = (player == 1) ? 'X' : 'O';

    while (1) {
        printf("Please enter number for Player %d: ", player);
        scanf("%d", &input);

        // Check for invalid input
        if (input >= 1 && input <= 9 && board[input] != 'X' && board[input] != 'O') {
            board[input] = mark;
            break;
        } else {
            printf("Invalid input. Please try again.\n");
        }
    }
}

int checkWin() {
    // Winning combinations
    int winCombos[8][3] = {
        {1, 2, 3}, {4, 5, 6}, {7, 8, 9}, // Rows
        {1, 4, 7}, {2, 5, 8}, {3, 6, 9}, // Columns
        {1, 5, 9}, {3, 5, 7}  // Diagonals
    };

    for (int i = 0; i < 8; i++) {
        if (board[winCombos[i][0]] == board[winCombos[i][1]] && board[winCombos[i][1]] == board[winCombos[i][2]]) {
            return 1;
        }
    }

    return 0;
}

int checkDraw() {
    for (int i = 1; i <= 9; i++) {
        if (board[i] != 'X' && board[i] != 'O') {
            return 0;  // Not a draw if there's an empty spot
        }
    }
    return 1;  // Draw if all spots are filled
}

void switchPlayer(int *player) {
    *player = (*player == 1) ? 2 : 1;
}
